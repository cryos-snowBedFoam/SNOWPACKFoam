203  template <class T> bool convertString(T& t, std::string str, std::ios_base& (*f)(std::ios_base&) = std::dec) {
204  trim(str); //delete trailing and leading whitespaces and tabs
205  if (str.empty()) {
206  t = static_cast<T> (nodata);
207  return true;
208  } else {
209  std::istringstream iss(str);
210  iss.setf(std::ios::fixed);
211  iss.precision(std::numeric_limits<double>::digits10); //try to read values with maximum precision
212  iss >> f >> t; //Convert first part of stream with the formatter (e.g. std::dec, std::oct)
213  if (iss.fail()) {
214  //Conversion failed
215  return false;
216  }
217  std::string tmp;
218  getline(iss, tmp); //get rest of line, if any
219  trim(tmp);
220  if (!tmp.empty() && tmp[0] != '#' && tmp[0] != ';') {
221  //if line holds more than one value it's invalid
222  return false;
223  }
224  return true;
225  }
226  }

245  template <class T> void getValueForKey(const std::map<std::string,std::string>& properties,
246  const std::string& key, T& t, const ThrowOptions& options=IOUtils::dothrow) {
247  if (key.empty() && options!=IOUtils::nothrow)
248  throw InvalidArgumentException("Empty key", AT);
249 
250  const std::map<std::string, std::string>::const_iterator it( properties.find(key) );
251  if (it == properties.end()){
252  if (options == IOUtils::nothrow)
253  return;
254  else
255  throw UnknownValueException("No value for key " + key, AT);
256  }
257  const std::string& value = it->second;
258 
259  if (!convertString<T>(t, value, std::dec) && options!=IOUtils::nothrow) {
260  std::cerr << "[E] When reading \"" << key << "\" = \"" << t << "\"\n";
261  throw ConversionFailedException(value, AT);
262  }
263  }

273  template <class T> void getValueForKey(const std::map<std::string,std::string>& properties,
274  const std::string& key, std::vector<T>& vecT, const ThrowOptions& options=IOUtils::dothrow)
275  {
276  if (key.empty() && options!=IOUtils::nothrow)
277  throw InvalidArgumentException("Empty key", AT);
278 
279  const std::map<std::string, std::string>::const_iterator it( properties.find(key) );
280  if (it == properties.end()) {
281  if (options == IOUtils::nothrow) {
282  return;
283  } else {
284  throw UnknownValueException("No value for key " + key, AT);
285  }
286  }
287  const std::string& value = it->second;
288 
289  //split value string
290  std::vector<std::string> vecUnconvertedValues;
291  const size_t counter = readLineToVec(value, vecUnconvertedValues);
292  for (size_t ii=0; ii<counter; ii++){
293  T myvar;
294  if (!convertString<T>(myvar, vecUnconvertedValues.at(ii), std::dec) && options!=IOUtils::nothrow){
295  std::cerr << "[E] When reading \"" << key << "\" = \"" << myvar << "\"\n";
296  throw ConversionFailedException(vecUnconvertedValues.at(ii), AT);
297  }
298  vecT.push_back(myvar);
299  }
300  }
